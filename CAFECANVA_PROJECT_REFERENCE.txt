import { Request, Response, NextFunction } from 'express';
import { db } from '../src/config/db.js';
import { sql } from 'drizzle-orm';

declare global {
  namespace Express {
    interface Request {
      user?: {
        id: string;
        tenant_id: string;
        branch_id: string | null;
        role: 'PLATFORM_ADMIN' | 'TENANT_OWNER' | 'BRANCH_ADMIN' | 'MANAGER' | 'STAFF' | 'KOS';
      };
      txQuery?: <T>(callback: (tx: any) => Promise<T>) => Promise<T>;
    }
  }
}

function decodeJWT(token: string) {
  try {
    const parts = token.split('.');
    if (parts.length !== 3) return null;
    const payloadJson = Buffer.from(parts[1], 'base64').toString('utf8');
    return JSON.parse(payloadJson);
  } catch (err) {
    return null;
  }
}

export async function storeAdminAuth(req: Request, res: Response, next: NextFunction) {
  try {
    const authHeader = req.headers.authorization;
    if (!authHeader || !authHeader.startsWith('Bearer ')) {
      // Fallback to dev headers
      const devTenantId = req.headers['x-tenant-id'] as string;
      const devBranchId = req.headers['x-branch-id'] as string;
      const devRole = req.headers['x-role'] as string;
      const devUserId = req.headers['x-user-id'] as string;

      if (devTenantId && devRole) {
        req.user = {
          id: devUserId || '00000000-0000-0000-0000-000000000000',
          tenant_id: devTenantId,
          branch_id: devBranchId || null,
          role: devRole as any
        };
      } else {
        res.status(401).json({ success: false, error: 'Unauthorized' });
        return;
      }
    } else {
      const token = authHeader.split(' ')[1];
      const payload = decodeJWT(token);

      if (!payload) {
        res.status(401).json({ success: false, error: 'Invalid JWT' });
        return;
      }

      const appMetadata = payload.app_metadata || {};
      const tenantId = appMetadata.tenant_id;
      const branchId = appMetadata.branch_id || null;
      const role = appMetadata.role;

      if (!tenantId || !role) {
        res.status(403).json({ success: false, error: 'Missing tenant_id or role' });
        return;
      }

      req.user = {
        id: payload.sub || '00000000-0000-0000-0000-000000000000',
        tenant_id: tenantId,
        branch_id: branchId,
        role: role
      };
    }

    const allowedRoles = ['TENANT_OWNER', 'BRANCH_ADMIN', 'MANAGER'];
    if (!allowedRoles.includes(req.user.role)) {
      res.status(403).json({ success: false, error: 'Insufficient privileges' });
      return;
    }

    // Inject Transaction Query Runner with RLS
    const user = req.user;
    req.txQuery = async <T>(callback: (tx: any) => Promise<T>): Promise<T> => {
      return await db.transaction(async (tx) => {
        await tx.execute(sql`SET LOCAL app.current_tenant_id = ${user.tenant_id}`);
        if (user.branch_id) {
          await tx.execute(sql`SET LOCAL app.current_branch_id = ${user.branch_id}`);
        }
        await tx.execute(sql`SET LOCAL app.current_role = ${user.role}`);
        return await callback(tx);
      });
    };

    next();
  } catch (err) {
    next(err);
  }
}

import { ipcMain, BrowserWindow, Notification } from 'electron';

export function registerIpcHandlers(mainWindow: BrowserWindow | null) {
  // Handle silent thermal receipt printing
  ipcMain.handle('print-receipt', async (event, htmlContent: string) => {
    if (!mainWindow) return { success: false, error: 'Main window not initialized' };
    try {
      const printWindow = new BrowserWindow({
        show: false,
        webPreferences: { nodeIntegration: true, contextIsolation: false },
      });
      await printWindow.loadURL(`data:text/html;charset=utf-8,${encodeURIComponent(htmlContent)}`);
      printWindow.webContents.print({
        silent: true, printBackground: true, margins: { marginType: 'none' },
      }, (success, errorType) => { printWindow.destroy(); });
      return { success: true };
    } catch (err: any) {
      return { success: false, error: err.message };
    }
  });

  // Handle local push notifications
  ipcMain.on('send-notification', (event, { title, body }) => {
    if (Notification.isSupported()) {
      new Notification({ title, body }).show();
    }
  });
}

import { type NextRequest, NextResponse } from 'next/server'
import { updateSession } from '@/utils/supabase/middleware'

export async function middleware(request: NextRequest) {
  const { pathname } = request.nextUrl

  // Skip static assets
  if (pathname.startsWith('/_next') || pathname.startsWith('/favicon.ico') ||
      pathname.match(/\.(?:svg|png|jpg|jpeg|gif|webp)$/)) {
    return NextResponse.next()
  }

  // Demo Mode Bypass
  const isDemoMode = process.env.NEXT_PUBLIC_DEMO_MODE === 'true'
  if (isDemoMode && pathname.startsWith('/demo')) return NextResponse.next()

  // Skip public webhooks
  if (pathname.startsWith('/api/payment/verify')) return NextResponse.next()

  // Session management for protected routes
  return await updateSession(request)
}

import { createServerClient } from '@supabase/ssr'
import { NextResponse, type NextRequest } from 'next/server'

export async function updateSession(request: NextRequest) {
  let supabaseResponse = NextResponse.next({ request })
  const supabase = createServerClient(
    process.env.NEXT_PUBLIC_SUPABASE_URL || 'https://placeholder.supabase.co',
    process.env.NEXT_PUBLIC_SUPABASE_PUBLISHABLE_KEY || 'placeholder',
    {
      cookies: {
        getAll() { return request.cookies.getAll() },
        setAll(cookiesToSet) {
          cookiesToSet.forEach(({ name, value }) => request.cookies.set(name, value))
          supabaseResponse = NextResponse.next({ request })
          cookiesToSet.forEach(({ name, value, options }) =>
            supabaseResponse.cookies.set(name, value, options))
        },
      },
    }
  )
  await supabase.auth.getUser()
  return supabaseResponse
}

  Source:    d:\Cafe Canva\.agents\skills\supabase\SKILL.md
  Version:   0.1.2
  Author:    Supabase

  PURPOSE:
    Comprehensive guidance for ALL Supabase tasks — Database, Auth,
    Edge Functions, Realtime, Storage, Vectors, Cron, Queues, client
    libraries (@supabase/ssr, supabase-js), CLI, and MCP server.

  TRIGGER CONDITIONS:
    - Working with Supabase products (Database, Auth, Edge Functions, etc.)
    - Using client libraries (supabase-js, @supabase/ssr) in Next.js/React
    - Auth issues (login, logout, sessions, JWT, cookies, RLS)
    - Supabase CLI or MCP server operations
    - Schema changes, migrations, security audits
    - Postgres extensions (pg_graphql, pg_cron, pg_vector)

  CORE PRINCIPLES:
    1. VERIFY AGAINST CHANGELOG: Supabase changes frequently.
       Fetch https://supabase.com/changelog.md before implementing.
    2. VERIFY YOUR WORK: Run test queries after every fix.
    3. DON'T LOOP: If approach fails 2-3 times, try a different method.
    4. DATA API EXPOSURE: New tables may need explicit GRANT for access.
    5. RLS EVERYWHERE: Enable RLS on every table in exposed schemas.
    6. SECURITY CHECKLIST:
       - Never use user_metadata in JWT authorization (user-editable!)
       - Deleting user doesn't invalidate existing tokens
       - Views bypass RLS by default (use security_invoker = true)
       - UPDATE requires a SELECT policy
       - auth.role() is DEPRECATED — use TO clause instead
       - TO authenticated alone = auth without authorization (BOLA/IDOR)
       - UPDATE policies need both USING and WITH CHECK
       - SECURITY DEFINER functions bypass RLS
       - Storage upsert requires INSERT + SELECT + UPDATE

  CLI COMMANDS:
    supabase --help                     # All top-level commands
    supabase <group> --help             # Subcommands
    supabase <group> <command> --help   # Flags for specific command

  CLI GOTCHAS:
    - supabase db query requires CLI v2.79.0+
    - supabase db advisors requires CLI v2.81.3+
    - Always create migrations with: supabase migration new <name>

  SCHEMA CHANGE WORKFLOW:
    1. Use execute_sql (MCP) or supabase db query to make changes
    2. Run advisors: supabase db advisors
    3. Review security checklist
    4. Generate migration: supabase db pull <name> --local --yes
    5. Verify: supabase migration list --local
